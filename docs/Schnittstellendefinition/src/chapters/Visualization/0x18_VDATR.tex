\subsubsection{Visualization Data Register (VDATR)}

Address offset: \textbf{0x18}
\newline
Reset value: 0x0000 0000

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader[lsb=16]{16-31} \\
        \bitbox{16}{Reserved}
    \end{bytefield}
\end{figure}

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader{0-15} \\
        \bitbox{7}{Reserved} &
        \bitbox{9}{VOL} \\
        \bitbox[]{7}{} &
        \bitbox[]{9}{R/W}
    \end{bytefield}
\end{figure}

\begin{itemize}
    \item \textbf{Bits 31:9 - Reserved}
    \item \textbf{Bits 8:0 - VOL: Volume}\\
    Enthält die zu visualisierende Lautstärke in einem 7.2 Festkommaformat in Prozent.
    Der geschriebene Wert darf 100 nicht überschreiten.
    Der Wert darf nur geschrieben werden, wenn das FDE Bit im IPISR Register 0 gelesen wird.
\end{itemize}
