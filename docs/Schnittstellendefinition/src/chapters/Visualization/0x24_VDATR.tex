\subsubsection{Visualization Data Register (VDATR)}

Address offset: \textbf{0x24}
\newline
Reset value: 0x0000 0000

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader[lsb=16]{16-31} \\
        \bitbox{16}{Reserved}
    \end{bytefield}
\end{figure}

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader{0-15} \\
        \bitbox{9}{Reserved} &
        \bitbox{7}{CHAR} \\
        \bitbox[]{9}{} &
        \bitbox[]{7}{R/W}
    \end{bytefield}
\end{figure}

\begin{itemize}
    \item \textbf{Bits 31:7 - Reserved}
    \item \textbf{Bits 6:0 - CHAR: Char}\\
    Das zu setzende Zeichen in ASCII an den in ADDRR angegebene Pixelblock.
    Wird übernommen, wenn WD in CTRL auf 1 gesetzt wird.
\end{itemize}
